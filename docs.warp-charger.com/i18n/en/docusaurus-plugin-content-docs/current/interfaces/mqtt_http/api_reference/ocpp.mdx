    <td>

    </td>
</tr>

<tr>
    <th scope="row" style={{whiteSpace: "nowrap"}}>`cert_id`<br/>int</th>
    <td>
        ID of the TLS certificate to be used for establishing a TLS-encrypted connection.

        - **>= 0** - Use certificate with this ID
    </td>
</tr>
